// --- 核心上傳邏輯優化版 --- async function processFile(file) { if (!file || !currentUser) return; const statusEl = document.getElementById('upload-status'); statusEl.innerHTML = `

正在深度解析檔案...

`; const reader = new FileReader(); reader.onload = async (ev) => { try { const csvContent = ev.target.result; const lines = csvContent.split(/\r?\n/); const records = []; // 更穩健的 CSV 分隔正則：處理引號內的逗號與空格 const csvRegex = /,(?=(?:(?:[^"]*"){2})*[^"]*$)/; // 從第 3 行開始 (index 3) 是數據起始點 for (let i = 3; i < lines.length; i++) { const line = lines[i].trim(); if (!line || line.replace(/,/g, '').trim() === '') continue; const cols = line.split(csvRegex).map(c => c.trim().replace(/^"|"$/g, '').trim() ); // 基礎欄位校驗：確保至少有姓名(1)與金額(8)與案別(9) if (cols.length < 10 || !cols[1]) continue; // 深度檢查狀態：掃描 10 欄位之後的所有欄位，確保抓到「已發完」 const remarkContent = cols.slice(10).join(' '); const isPaid = remarkContent.includes('已發完') || remarkContent.includes('已領'); records.push({ date: cols[0], name: cols[1], role: cols[2], // 存入前先移除千分撇，確保運算安全 amount: cols[8].replace(/[^0-9.-]+/g, ''), project: cols[9], status: isPaid ? '已領' : '未領', createdAt: Date.now() }); } if (records.length === 0) { throw new Error("找不到有效的獎金資料，請檢查 CSV 格式。"); } // 批次寫入 Firebase (每 400 筆一包) const batchSize = 400; for (let i = 0; i < records.length; i += batchSize) { const batch = writeBatch(db); const chunk = records.slice(i, i + batchSize); statusEl.innerHTML = `

寫入雲端中 (${i}/${records.length})...

`; chunk.forEach(record => { const ref = doc(collection(db, 'artifacts', appId, 'public', 'data', 'bonuses')); batch.set(ref, record); }); await batch.commit(); } alert(`🎉 成功解析並匯入 ${records.length} 筆獎金紀錄！`); } catch (err) { console.error("解析失敗:", err); alert(`讀取失敗：${err.message}`); } finally { resetUploadStatus(); } }; // 解決台灣 Excel Big5 編碼問題：若亂碼可改用 'Big5' reader.readAsText(file, 'UTF-8'); }
